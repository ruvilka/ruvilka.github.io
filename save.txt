double ost(int count, double value) {
    double res = value / count; //1.5
    int s = res; //1
    double w = res - s; //1.5 - 1 = 0.5
    return w;
}

void n26 ()
{
    cout << "Задание 26:" << endl;
    double a;   
    double b;   
    double c;   
    cout << "ax2 + bx + c = 0\na: ";
    cin >> a;
    cout << "b: ";
    cin >> b;
    cout << "c: ";
    cin >> c;
    double disc = b*b - 4 * a * c;
    if (disc < 0){
        cout << "меньше 0\nкорней нет" << endl;
        
    }
    else if (disc == 0){
        cout << "равно 0, 1 корень: " << -b/(2*a) << endl;
        
    }
    else{
        cout << "больше 0, 2 корня:\n1) " << ((0-b)+ pow(disc, 0.5))/2*a << "\n2) " << ((0-b)- pow(disc, 0.5))/2*a << endl;
        
    }

}
